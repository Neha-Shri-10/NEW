#include <stdio.h>
#include<stdlib.h>
int main() {
    int n,i,j,index=-1;
    printf("Enter the n :");
    scanf("%d",&n);
    int *pet=(int*)malloc(n*sizeof(int));
    int *dist=(int*)malloc(n*sizeof(int));
    printf("Petrol:");
    for(i=0;i<n;i++){
        scanf("%d",&pet[i]);
    }
    printf("Distance:");
    for(i=0;i<n;i++){
        scanf("%d",&dist[i]);
    }
    printf("Petrol pump pairs:");
    for(i=0,j=0;i<n,j<n;i++,j++){
        printf("{ %d , %d }",pet[i],dist[j]);
    }
    int sumpet=0,sumdist=0;
    for(i=0;i<n;i++){
        sumpet+=pet[i];
        sumdist+=dist[i];
    }
    if(sumpet<sumdist){
        index=-1;
    }
    int share=0,curr=0;
    for(i=0;i<n;i++){
        //share+=pet[i]-dist[i];
        curr+=pet[i]-dist[i];
        if(curr<0){
            share++;
            curr=0;
        }
    }
    printf("Output: %d",share);
    
    }
