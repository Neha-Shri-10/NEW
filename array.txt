#include <stdio.h>

void solve() {
    int n;
    if (scanf("%d", &n) != 1) return;
    
    int low = n;
    int high = 2 * n;
    
    // Generate and print the interleaved array sequences
    for (int j = 0; j < n; j++) {
        if (j % 2 == 0) {
            printf("%d", low);
            low--;
        } else {
            printf("%d", high);
            high--;
        }
        
        if (j < n - 1) {
            printf(" ");
        } else {
            printf("\n");
        }
    }
}

int main() {
    int t;
    if (scanf("%d", &t) == 1) {
        while (t--) {
            solve();
        }
    }
    return 0;
}
