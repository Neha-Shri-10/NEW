#include <stdio.h>
int main() {
    int decimal, temp;
    int binary[32];     int i = 0, count = 0;

    if (scanf("%d", &decimal) != 1) {
        printf("invalid input\n");
        return 0;
    }
    if (decimal <= 0) {
        printf("invalid input\n");
        return 0;
    }
    temp = decimal;
    while (temp > 0) {
        binary[i] = temp % 2;
        if (binary[i] == 1) {
            count++;
        }
        temp = temp / 2;
        i++;
    }
    if (count == 0) {
        printf("invalid input\n");
        return 0;
    }

    for (int j = i - 1; j >= 0; j--) {
        printf("%d", binary[j]);
    }

    printf("\n%d\n", count);

    return 0;
}
